\begin{context}
    \textbf{Dựa vào thông tin sau để trả lời các câu hỏi 91, 92, 93:} Một biên tập viên sẽ biên tập 7 bài viết, mỗi lần một bài: có 3 bài viết $G, H$ và $J$ thuộc chủ đề tài chính; 3 bài viết khác $Q, R$ và $S$ thuộc chủ đề dinh dưỡng; và bài viết còn lại là $Y$ thuộc chủ đề động vật hoang dã. Thứ tự biên tập phải tuân theo các điều kiện sau:
    \begin{itemize}
        \item Các bài viết liên tiếp không được thuộc cùng một chủ đề.
        \item $S$ chỉ có thể được biên tập trước $Q$ nếu $Q$ là bài viết thứ ba được biên tập.
        \item $S$ phải được biên tập trước $Y$ ($S < Y$).
        \item $J$ phải được biên tập trước $G$, và $G$ phải được biên tập trước $R$ ($J < G < R$).
    \end{itemize}
\end{context}

\begin{exam}[resume]
    \item Một thứ tự biên tập các bài viết thỏa mãn các yêu cầu nêu trên là:
    \begin{tasks}(1)
        \task $Q, J, Y, S, G, R, H$.
        \task $Q, J, S, H, Y, G, R$.
        \task $J, Q, G, H, S, Y, R$.
        \task $J, Q, S, Y, G, R, H$.
    \end{tasks}

    \item Nếu $G$ là bài viết thứ tư được biên tập, khẳng định nào sau đây \textbf{có thể đúng}?
    \begin{tasks}(2)
        \task $Q$ là bài viết thứ hai được biên tập.
        \task $Y$ là bài viết thứ sáu được biên tập.
        \task $S$ là bài viết thứ năm được biên tập.
        \task $J$ là bài viết thứ nhất được biên tập.
    \end{tasks}

    \item Thứ tự biên tập 7 bài viết có thể được xác định \textbf{hoàn toàn} nếu như:
    \begin{tasks}(2)
        \task $H$ là bài viết thứ tư được biên tập.
        \task $R$ là bài viết thứ tư được biên tập.
        \task $H$ là bài viết thứ sáu được biên tập.
        \task $R$ là bài viết thứ bảy được biên tập.
    \end{tasks}
\end{exam} % ĐÓNG CÂU 91-93 TẠI ĐÂY

\begin{context}

\begin{context}
    \textbf{Dựa vào thông tin dưới đây để trả lời các câu từ 103 đến 105:} 
    
    Các chất có khuynh hướng tách khỏi bề mặt của chất lỏng để chuyển thành trạng thái hơi và gây áp suất hơi. Khi tốc độ bay hơi bằng tốc độ ngưng tụ thì áp suất hơi lúc đó gọi là áp suất hơi bão hòa. Áp suất hơi bão hòa tỷ lệ thuận với nhiệt độ. Ở nhiệt độ mà áp suất hơi bão hòa bằng áp suất khí quyển thì nhiệt độ đó chính là nhiệt độ sôi của chất lỏng.

    Kết quả xác định áp suất hơi bão hòa của 7 hỗn hợp gồm $n$-heptane và $n$-octane ở $25^\circ C$ được thể hiện trong bảng sau:

        2 & 0,2 & 0,8 & 20,4 \\ \hline
        3 & 0,4 & 0,6 & 26,8 \\ \hline
        4 & 0,5 & 0,5 & 30,0 \\ \hline

Nếu năm 2024, GRDP của tỉnh chỉ tăng trưởng $4,07\%$, thấp hơn trung bình cả nước là $8,02\%$ thì năm 2025 địa phương này vượt lên nhóm dẫn đầu nhờ tăng trưởng mạnh trong sản xuất công nghiệp, xây dựng và dịch vụ. Sản phẩm lọc hóa dầu tăng $20\%$, giá trị sản xuất ngành thép tăng gần $35\%$

\begin{context}
    \textbf{Dựa vào thông tin dưới đây để trả lời các câu từ 109 đến 111:} 
    
   Máy biến áp là một thiết bị có khả năng biến đổi điện áp sử dụng nguyên lý cảm ứng điện từ. Điện áp xoay chiều được đưa vào hai đầu cuộn dây sơ cấp, dòng điện này sẽ sinh ra từ trường biến thiên trong cuộn sơ cấp và từ trường biến thiên này theo lõi thép được chuyển sang cuộn thứ cấp. Từ trường biến thiên trong cuộn thứ cấp sẽ làm sinh ra sức điện động cảm ứng và từ đó tạo ra điện áp ở hai đầu cuộn thứ cấp $U_2$. Trong trường hợp lý tưởng, tỷ lệ điện áp trên cuộn thứ cấp $U_2$ và điện áp cuộn sơ cấp $U_1$ bằng với tỷ lệ số vòng dây trên cuộn thứ cấp $N_2$ so với trên cuộn sơ cấp $N_1$ theo công thức:
\begin{equation}
\frac{U_2}{U_1} = \frac{N_2}{N_1} \text{}
\end{equation}

\begin{center}
\begin{tikzpicture}[scale=0.8]
    % Lõi thép
    \draw[thick, fill=gray!20] (0,0) rectangle (5,4);
    \draw[thick, fill=white] (1,1) rectangle (4,3);
    \node at (2.5, 3.5) {Lõi thép};

    % Cuộn sơ cấp (N1)
    \draw[ultra thick, blue] (-0.5, 3.2) -- (0.5, 3.2);
    \foreach \i in {0,1,2,3,4} {
        \draw[ultra thick, blue] (0, 3.2 - \i*0.5) arc (90:270:0.2 and 0.25);
    }
    \draw[ultra thick, blue] (-0.5, 0.8) -- (0.5, 0.8);
    \node[left] at (-0.5, 2) {$U_1$};
    \node[below] at (0, 0.5) {$N_1$};
    \node[below] at (0, 0) {Cuộn sơ cấp};

    % Cuộn thứ cấp (N2)
    \draw[ultra thick, red] (4.5, 3.2) -- (5.5, 3.2);
    \foreach \i in {0,1,2,3,4,5,6} {
        \draw[ultra thick, red] (5, 3.2 - \i*0.35) arc (90:-90:0.2 and 0.175);
    }
    \draw[ultra thick, red] (4.5, 0.8) -- (5.5, 0.8);
    \node[right] at (5.5, 2) {$U_2$};
    \node[below] at (5, 0.5) {$N_2$};
    \node[below] at (5, 0) {Cuộn thứ cấp};
\end{tikzpicture}
\end{center}

Tuy nhiên trong quá trình hoạt động, chính hiện tượng cảm ứng điện từ cũng sinh ra dòng điện cảm ứng bên trong lõi thép, có thể dẫn tới việc tỏa nhiệt mạnh, gây hao phí, biến dạng hoặc cháy nổ. Do đó, người ta chế tạo lõi thép bằng các lá thép mỏng cách điện với nhau để hạn chế dòng điện này. Trong truyền tải điện năng, việc tăng điện áp tại nguồn lên $n$ lần sẽ giảm hao phí truyền tải đi $n^2$ lần. Sử dụng một máy biến áp lý tưởng có số vòng dây cuộn sơ cấp là 10.000 vòng, cuộn thứ cấp là 50.000 vòng.

\end{context}
